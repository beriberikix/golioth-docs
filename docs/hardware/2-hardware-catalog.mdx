import boards from "./assets/boards.json";
import Image from "@theme/IdealImage";
import mag from "./assets/img/magnifier.svg";
import "./assets/hardware-catalog.css";
import { useState, useEffect } from "react";
import { FontAwesomeIcon } from "@fortawesome/react-fontawesome";
import {
  faCheck,
  faSearch,
  faUserCheck,
  faCheckDouble,
  faSearchMinus,
  faQuestion,
} from "@fortawesome/free-solid-svg-icons";

export const BoardStatusBadge = ({ goliothStatus }) => {
  switch (goliothStatus) {
    case "quickstart":
      return (
        <FontAwesomeIcon
          icon={faCheckDouble}
          className="board-status-badge green"
        />
      );
    case "verified":
      return (
        <FontAwesomeIcon icon={faCheck} className="board-status-badge green" />
      );
    case "community":
      return (
        <FontAwesomeIcon
          icon={faUserCheck}
          className="board-status-badge blue"
        />
      );
    default:
      return (
        <FontAwesomeIcon icon={faQuestion} className="board-status-badge" />
      );
  }
};

export const Board = ({ board }) => {
  const { boardId, name, img, goliothStatus } = board;
  return (
    <div className="board-card">
      <div className="board-card-image">
        <div className="board-card-badge-container">
          <BoardStatusBadge goliothStatus={goliothStatus} />
        </div>
        <Image img={require(`./assets/${img}`)} />
      </div>
      <div className="board-card-footer">{name}</div>
    </div>
  );
};

export const SearchInput = ({ boards, onSearch }) => {
  const [search, setSearch] = useState("");
  useEffect(() => {
    onSearch(search);
  }, [search]);
  return (
    <div className="search-input-group">
      <span className="search-icon">
        <FontAwesomeIcon icon={faSearch} />
      </span>
      <input
        className="search-input"
        type="search"
        results="5"
        placeholder="Search by board name"
        value={search}
        onChange={(e) => setSearch(e.target.value)}
      />
    </div>
  );
};

export const Legend = () => {
  return (
    <div className="legend">
      <div className="legend-title">Legend:</div>
      <div className="legend-item">
        <FontAwesomeIcon
          icon={faCheckDouble}
          className="board-status-badge green"
        />
        <span>Verified + Quickstart</span>
      </div>
      <div className="legend-item">
        <FontAwesomeIcon icon={faCheck} className="board-status-badge green" />
        <span>Verified</span>
      </div>
      <div className="legend-item">
        <FontAwesomeIcon
          icon={faUserCheck}
          className="board-status-badge blue"
        />
        <span>Community</span>
      </div>
      <div className="legend-item">
        <FontAwesomeIcon icon={faQuestion} className="board-status-badge" />
        <span>Unverified</span>
      </div>
    </div>
  );
};

export const Empty = () => {
  return (
    <div className="empty">
      <span className="empty-icon">
        <FontAwesomeIcon icon={faSearchMinus} />
      </span>
      <div className="empty-text">No boards found</div>
      <div className="empty-detail">Check your active filters</div>
    </div>
  );
};

export const BoardList = ({ boards }) => {
  const sortOrder = {
    quickstart: 1,
    verified: 2,
    community: 3,
    unverified: 4,
  };
  const [query, setQuery] = useState("");
  const [filteredBoards, setFilteredBoards] = useState(boards);
  const [sortedBoards, setSortedBoards] = useState(boards);
  //debounce the search
  useEffect(() => {
    const timeout = setTimeout(() => {
      const filtered = boards.filter(({ name }) =>
        name.toLowerCase().includes(query.toLowerCase()),
      );
      setFilteredBoards(filtered);
    }, 500);
    return () => clearTimeout(timeout);
  }, [query]);
  //sort filtered boards by sort order
  useEffect(() => {
    setSortedBoards(
      filteredBoards.sort((a, b) => {
        return sortOrder[a.goliothStatus] - sortOrder[b.goliothStatus];
      })
    );
  }, [filteredBoards]);
  return (
    <>
      <SearchInput boards={boards} onSearch={setQuery} />
      <Legend />
      <div className="board-list">
        {sortedBoards.map((board) => (
          <Board key={board.boardId} board={board} />
        ))}
        {sortedBoards.length === 0 && <Empty />}
      </div>
    </>
  );
};

<BoardList boards={boards} />
